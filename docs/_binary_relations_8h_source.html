11template <typename T> int countInSortedVector(const std::vector<T> *vector, const T &value)
12{
13 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
14 return it != vector->cend() && *it == value ? 1 : 0;
15}
16
17template <typename T>
18typename std::vector<T>::const_iterator findInSortedVector(const std::vector<T> *vector, const T &value)
19{
20 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
21 if (it != vector->cend() && *it == value)
22 return it;
23 else
24 return vector->cend();
25}
26
27template <typename T> int insertIntoSortedVector(std::vector<T> *vector, const T &value)
28{
29 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
30 if (it != vector->end() && *it == value)
31 return 0; // It's already there
32 else
33 {
34 vector->insert(it, value);
35 return 1;
36 }
37}
38
39template <typename T> int removeFromSortedVector(std::vector<T> *vector, const T &value)
40{
41 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
42 if (it != vector->end() && *it == value)
43 {
44 vector->erase(it);
45 return 1;
46 }
47 else
48 return 0;
49}

51template <typename T>
52void insertIntoSortedVector(std::vector<T> *sourceVector, std::vector<T> *insertVector, std::vector<T> *outVector)
53{
54 typename std::vector<T>::const_iterator sourceIt = sourceVector->cbegin();
55 typename std::vector<T>::const_iterator insertIt = insertVector->cbegin();
56
57 outVector->clear();
58 outVector->reserve(sourceVector->count() + insertVector->count());
59
60 while (sourceIt != sourceVector->cend() && insertIt != insertVector->cend())
61 {
62 if (*sourceIt == *insertIt)
63 {
64 outVector->append(*sourceIt++);
65 insertIt++;
66 }
67 else if (*sourceIt < *insertIt)
68 outVector->append(*sourceIt++);
69 else
70 outVector->append(*insertIt++);
71 }
72
73 while (sourceIt != sourceVector->cend())
74 outVector->append(*sourceIt++);
75
76 while (insertIt != insertVector->cend())
77 outVector->append(*insertIt++);
78}
79
80template <typename T>
81void removeFromSortedVector(std::vector<T> *sourceVector, std::vector<T> *removeVector, std::vector<T> *outVector)
82{
83 typename std::vector<T>::const_iterator sourceIt = sourceVector->cbegin();
84 typename std::vector<T>::const_iterator removeIt = removeVector->cbegin();
85
86 outVector->clear();
87 outVector->reserve(std::max(sourceVector->count(), removeVector->count()));
88
89 while (sourceIt != sourceVector->cend() && removeIt != removeVector->cend())
90 {
91 if (*sourceIt == *removeIt)
92 {
93 sourceIt++;
94 removeIt++;
95 }
96 else if (*sourceIt < *removeIt)
97 outVector->append(*sourceIt++);
98 else
99 outVector->append(*removeIt++);
100 }
101
102 while (sourceIt != sourceVector->cend())
103 outVector->append(*sourceIt++);
104}
