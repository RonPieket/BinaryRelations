37// -------- Manipulate vector with unique sorted elements --------
38
39template <typename T> int countInSortedVector(const std::vector<T> *vector, const T &value) noexcept
40{
41 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
42 return it != vector->cend() && *it == value ? 1 : 0;
43}
44
45template <typename T>
46typename std::vector<T>::const_iterator findInSortedVector(const std::vector<T> *vector, const T &value) noexcept
47{
48 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
49 if (it != vector->cend() && *it == value)
50 return it;
51 else
52 return vector->cend();
53}
54
55template <typename T> int insertIntoSortedVector(std::vector<T> *vector, const T &value) noexcept
56{
57 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
58 if (it != vector->end() && *it == value)
59 return 0; // It's already there
60 else
61 {
62 vector->insert(it, value);
63 return 1;
64 }
65}
66
67template <typename T> int removeFromSortedVector(std::vector<T> *vector, const T &value) noexcept
68{
69 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
70 if (it != vector->end() && *it == value)
71 {
72 vector->erase(it);
73 return 1;
74 }
75 else
76 return 0;
77}
78
79template <typename T>
80void insertIntoSortedVector(std::vector<T> *sourceVector, std::vector<T> *insertVector, std::vector<T> *outVector) noexcept
81{
82 typename std::vector<T>::const_iterator sourceIt = sourceVector->cbegin();
83 typename std::vector<T>::const_iterator insertIt = insertVector->cbegin();
84
85 outVector->clear();
86 outVector->reserve(sourceVector->count() + insertVector->count());
87
88 while (sourceIt != sourceVector->cend() && insertIt != insertVector->cend())
89 {
90 if (*sourceIt == *insertIt)
91 {
92 outVector->append(*sourceIt++);
93 insertIt++;
94 }
95 else if (*sourceIt < *insertIt)
96 outVector->append(*sourceIt++);
97 else
98 outVector->append(*insertIt++);
99 }
100
101 while (sourceIt != sourceVector->cend())
102 outVector->append(*sourceIt++);
103
104 while (insertIt != insertVector->cend())
105 outVector->append(*insertIt++);
106}
107
108template <typename T>
109void removeFromSortedVector(std::vector<T> *sourceVector, std::vector<T> *removeVector, std::vector<T> *outVector) noexcept
110{
111 typename std::vector<T>::const_iterator sourceIt = sourceVector->cbegin();
112 typename std::vector<T>::const_iterator removeIt = removeVector->cbegin();
113
114 outVector->clear();
115 outVector->reserve(std::max(sourceVector->count(), removeVector->count()));
116
117 while (sourceIt != sourceVector->cend() && removeIt != removeVector->cend())
118 {
119 if (*sourceIt == *removeIt)
120 {
121 sourceIt++;
122 removeIt++;
123 }
124 else if (*sourceIt < *removeIt)
125 outVector->append(*sourceIt++);
126 else
127 outVector->append(*removeIt++);
128 }
129
130 while (sourceIt != sourceVector->cend())
131 outVector->append(*sourceIt++);
132}

196{
197 std::unordered_map<LeftType, std::vector<RightType> *> m_LeftToRight;
198 std::unordered_map<RightType, LeftType> m_RightToLeft;
199 std::vector<RightType> m_EmptyRightVector;

201public:
205 struct Pair
206 {
207 LeftType left;
208 RightType right;
209 Pair(LeftType left, RightType right)
210 : left(left), right(right)

213 {}

238 void insert(const LeftType &left, const RightType &right) noexcept
239 {
240 auto r2l_it = m_RightToLeft.find(right);
241 if (r2l_it != m_RightToLeft.end())
242 {
243 if (r2l_it->second == left)
244 return; // We already have this pair - do nothing
245
246 remove(r2l_it->second, right); // Remove old relation
247 }
248
249 auto l2r_vec = m_LeftToRight[left]; // Will insert if it isn't already there.
250 if (l2r_vec == nullptr)
251 {
252 l2r_vec = new std::vector<RightType>();
253 m_LeftToRight[left] = l2r_vec;
254 }
255 insertIntoSortedVector(l2r_vec, right);
256
257 m_RightToLeft[right] = left;
258 }

266 {
267 for (auto pair : other)
268 {
269 insert(pair);
270 }
271 }

278 void remove(const Pair &pair) noexcept
279 {
280 remove(pair.left, pair.right);
281 }
282
289 void remove(const LeftType &left, const RightType &right) noexcept
290 {
291 auto r2l_it = m_RightToLeft.find(right);
292 if (r2l_it == m_RightToLeft.end())
293 return; // (*,right) not in the set - do nothing
294
295 if (r2l_it->second != left)
296 return; // (left,right) is not in the set - do nothing
297
298 auto l2r_it = m_LeftToRight.find(left);
299 auto l2r_vec = l2r_it->second;
300 auto l2r_vec_it = findInSortedVector(l2r_vec, right);
301 if (l2r_vec_it != l2r_vec->cend())
302 {
303 l2r_vec->erase(l2r_vec_it);
304 if (l2r_vec->size() == 0)
305 {
306 m_LeftToRight.erase(l2r_it);
307 delete l2r_vec; // Vector is empty now
308 }
309 m_RightToLeft.erase(r2l_it);
310 }
311 }
312
318 void removeLeft(const LeftType &left) noexcept
319 {
320 auto l2r_it = m_LeftToRight.find(left);
321 if (l2r_it == m_LeftToRight.end())
322 return;
323
324 auto l2r_vec = l2r_it->second;
325 for (auto right : l2r_vec)
326 {
327 auto r2l_it = m_RightToLeft.find(right);
328 m_RightToLeft.erase(r2l_it);
329 }
330
331 m_LeftToRight.erase(l2r_it);
332 delete l2r_vec; // Vector is empty now
333 }
334
340 void removeRight(const RightType &right) noexcept
341 {
342 auto r2l_it = m_RightToLeft.find(right);
343 if (r2l_it == m_RightToLeft.end())
344 return;
345 remove(r2l_it->second, right);
346 }

354 for (auto pair : other)
355 {
356 remove(pair);
357 }
358 }

563template <typename LeftType, typename RightType> class ManyToMany
564{
565 std::unordered_map<LeftType, std::vector<RightType> *> m_LeftToRight;
566 std::unordered_map<RightType, std::vector<LeftType> *> m_RightToLeft;
567 std::vector<RightType> m_EmptyRightVector;
568 std::vector<LeftType> m_EmptyLeftVector;
569 int m_Count;
570
571public:
575 struct Pair
576 {
577 LeftType left;
578 RightType right;
579 Pair(LeftType left, RightType right) : left(left), right(right)
580 {
581 }
582 Pair()
583 {
584 }
585 };
586
590 ManyToMany() noexcept
591 : m_Count(0)
592 {}
593
598 void insert(const Pair &pair) noexcept
599 {
600 insert(pair.left, pair.right);
601 }
602
608 void insert(const LeftType &left, const RightType &right) noexcept
609 {
610 auto l2r_it = m_LeftToRight.find(left);
611 if (l2r_it != m_LeftToRight.end())
612 {
613 auto l2r_vec = l2r_it->second;
614 if (0 != countInSortedVector(l2r_vec, right))
615 return; // We already have this pair;
616 }
617
618 auto l2r_vec = m_LeftToRight[left]; // Will insert if it isn't already there.
619 if (l2r_vec == nullptr)
620 {
621 l2r_vec = new std::vector<RightType>();
622 m_LeftToRight[left] = l2r_vec;
623 }
624 insertIntoSortedVector(l2r_vec, right);
625
626 auto r2l_vec = m_RightToLeft[right]; // Will insert if it isn't already there.
627 if (r2l_vec == nullptr)
628 {
629 r2l_vec = new std::vector<LeftType>();
630 m_RightToLeft[right] = r2l_vec;
631 }
632 insertIntoSortedVector(r2l_vec, left);
633 m_Count += 1;
634 }
635
640 void insert(const ManyToMany<LeftType, RightType>& other) noexcept
641 {
642 for (auto pair : other)
643 {
644 insert(pair);
645 }
646 }

653 void remove(const Pair &pair) noexcept
654 {
655 remove(pair.left, pair.right);
656 }
657
664 void remove(const LeftType &left, const RightType &right) noexcept
665 {
666 auto l2r_it = m_LeftToRight.find(left);
667 if (l2r_it != m_LeftToRight.end())
668 {
669 auto r2l_it = m_RightToLeft.find(right);
670 if (r2l_it != m_RightToLeft.end())
671 {
672 auto l2r_vec = l2r_it->second;
673 auto r2l_vec = r2l_it->second;
674
675 removeFromSortedVector(l2r_vec, right);
676 removeFromSortedVector(r2l_vec, left);
677
678 if(l2r_vec->size() == 0)
679 {
680 m_LeftToRight.erase(l2r_it);
681 delete l2r_vec;
682 }
683
684 if(r2l_vec->size() == 0)
685 {
686 m_RightToLeft.erase(r2l_it);
687 delete r2l_vec;
688 }
689
690 m_Count -= 1;
691 }
692 }
693 }
694
700 void removeLeft(const LeftType &left) noexcept
701 {
702 auto l2r_it = m_LeftToRight.constFind(left);
703 if (l2r_it != m_LeftToRight.end())
704 {
705 auto l2r_vec = l2r_it->second;
706 for (auto right : l2r_vec)
707 {
708 auto r2l_it = m_RightToLeft.constFind(right);
709 auto r2l_vec = r2l_it->second;
710 removeFromSortedVector(*r2l_vec, left);
711 m_Count -= 1;
712 if (r2l_vec->size() == 0)
713 {
714 m_RightToLeft.erase(r2l_it);
715 delete r2l_vec;
716 }
717 }
718 m_LeftToRight.erase(l2r_it);
719 delete l2r_vec;
720 }
721 }
722
728 void removeRight(const RightType &right) noexcept
729 {
730 auto r2l_it = m_RightToLeft.constFind(right);
731 if (r2l_it != m_RightToLeft.end())
732 {
733 auto r2l_vec = r2l_it->second;
734 for (auto left : r2l_vec)
735 {
736 auto l2r_it = m_LeftToRight.constFind(left);
737 auto l2r_vec = l2r_it->second;
738 removeFromSortedVector(*l2r_vec, right);
739 m_Count -= 1;
740 if (l2r_vec->size() == 0)
741 {
742 m_LeftToRight.erase(l2r_it);
743 delete l2r_vec;
744 }
745 }
746 m_RightToLeft.erase(r2l_it);
747 delete r2l_vec;
748 }
749 }
