37// -------- Manipulate vector with unique sorted elements --------
38
39template <typename T> int countInSortedVector(const std::vector<T> *vector, const T &value) noexcept
40{
41 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
42 return it != vector->cend() && *it == value ? 1 : 0;
43}
44
45template <typename T>
46typename std::vector<T>::const_iterator findInSortedVector(const std::vector<T> *vector, const T &value) noexcept
47{
48 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
49 if (it != vector->cend() && *it == value)
50 return it;
51 else
52 return vector->cend();
53}
54
55template <typename T> int insertIntoSortedVector(std::vector<T> *vector, const T &value) noexcept
56{
57 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
58 if (it != vector->end() && *it == value)
59 return 0; // It's already there
60 else
61 {
62 vector->insert(it, value);
63 return 1;
64 }
65}
66
67template <typename T> int removeFromSortedVector(std::vector<T> *vector, const T &value) noexcept
68{
69 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
70 if (it != vector->end() && *it == value)
71 {
72 vector->erase(it);
73 return 1;
74 }
75 else
76 return 0;
77}
78
79template <typename T>
80void insertIntoSortedVector(std::vector<T> *sourceVector, std::vector<T> *insertVector, std::vector<T> *outVector) noexcept
81{
82 typename std::vector<T>::const_iterator sourceIt = sourceVector->cbegin();
83 typename std::vector<T>::const_iterator insertIt = insertVector->cbegin();
84
85 outVector->clear();
86 outVector->reserve(sourceVector->count() + insertVector->count());
87
88 while (sourceIt != sourceVector->cend() && insertIt != insertVector->cend())
89 {
90 if (*sourceIt == *insertIt)
91 {
92 outVector->append(*sourceIt++);
93 insertIt++;
94 }
95 else if (*sourceIt < *insertIt)
96 outVector->append(*sourceIt++);
97 else
98 outVector->append(*insertIt++);
99 }
100
101 while (sourceIt != sourceVector->cend())
102 outVector->append(*sourceIt++);
103
104 while (insertIt != insertVector->cend())
105 outVector->append(*insertIt++);
106}
107
108template <typename T>
109void removeFromSortedVector(std::vector<T> *sourceVector, std::vector<T> *removeVector, std::vector<T> *outVector) noexcept
110{
111 typename std::vector<T>::const_iterator sourceIt = sourceVector->cbegin();
112 typename std::vector<T>::const_iterator removeIt = removeVector->cbegin();
113
114 outVector->clear();
115 outVector->reserve(std::max(sourceVector->count(), removeVector->count()));
116
117 while (sourceIt != sourceVector->cend() && removeIt != removeVector->cend())
118 {
119 if (*sourceIt == *removeIt)
120 {
121 sourceIt++;
122 removeIt++;
123 }
124 else if (*sourceIt < *removeIt)
125 outVector->append(*sourceIt++);
126 else
127 outVector->append(*removeIt++);
128 }
129
130 while (sourceIt != sourceVector->cend())
131 outVector->append(*sourceIt++);
132}

193template <typename LeftType, typename RightType> class OneToMany
194{
195 std::unordered_map<LeftType, std::vector<RightType> *> m_LeftToRight;
196 std::unordered_map<RightType, LeftType> m_RightToLeft;
197 std::vector<RightType> m_EmptyRightVector;
198
199public:
203 struct Pair
204 {
205 LeftType left;
206 RightType right;
207 Pair(LeftType left, RightType right)
208 : left(left), right(right)
209 {}
210 Pair()
211 {}
212 };

217 OneToMany() noexcept
218 {}

225 void insert(const Pair &pair) noexcept
226 {

236 void insert(const LeftType &left, const RightType &right) noexcept
237 {
238 auto r2l_it = m_RightToLeft.find(right);
239 if (r2l_it != m_RightToLeft.end())
240 {
241 if (r2l_it->second == left)
242 return; // We already have this pair - do nothing
243
244 remove(r2l_it->second, right); // Remove old relation
245 }
246
247 auto l2r_vec = m_LeftToRight[left]; // Will insert if it isn't already there.
248 if (l2r_vec == nullptr)
249 {
250 l2r_vec = new std::vector<RightType>();
251 m_LeftToRight[left] = l2r_vec;
252 }
253 insertIntoSortedVector(l2r_vec, right);
254
255 m_RightToLeft[right] = left;
256 }
257
263 void insert(const OneToMany<LeftType, RightType>& other) noexcept
264 {
265 for (auto pair : other)
266 {
267 insert(pair);
268 }
269 }
270
276 void remove(const Pair &pair) noexcept
277 {
278 remove(pair.left, pair.right);
279 }
280
287 void remove(const LeftType &left, const RightType &right) noexcept
288 {
289 auto r2l_it = m_RightToLeft.find(right);
290 if (r2l_it == m_RightToLeft.end())
291 return; // (*,right) not in the set - do nothing
292
293 if (r2l_it->second != left)
294 return; // (left,right) is not in the set - do nothing
295
296 auto l2r_it = m_LeftToRight.find(left);
297 auto l2r_vec = l2r_it->second;
298 auto l2r_vec_it = findInSortedVector(l2r_vec, right);
299 if (l2r_vec_it != l2r_vec->cend())
300 {
301 l2r_vec->erase(l2r_vec_it);
302 if (l2r_vec->size() == 0)
303 {
304 m_LeftToRight.erase(l2r_it);
305 delete l2r_vec; // Vector is empty now
306 }
307 m_RightToLeft.erase(r2l_it);
308 }
309 }
310
316 void removeLeft(const LeftType &left) noexcept
317 {
318 auto l2r_it = m_LeftToRight.find(left);
319 if (l2r_it == m_LeftToRight.end())
320 return;
321
322 auto l2r_vec = l2r_it->second;
323 for (auto right : l2r_vec)
324 {
325 auto r2l_it = m_RightToLeft.find(right);
326 m_RightToLeft.erase(r2l_it);
327 }
328
329 m_LeftToRight.erase(l2r_it);
330 delete l2r_vec; // Vector is empty now
331 }
332
338 void removeRight(const RightType &right) noexcept
339 {
340 auto r2l_it = m_RightToLeft.find(right);
341 if (r2l_it == m_RightToLeft.end())
342 return;
343 remove(r2l_it->second, right);
344 }
345
350 void remove(const OneToMany<LeftType, RightType> &other) noexcept
351 {
352 for (auto pair : other)
353 {
354 remove(pair);
355 }
356 }
357

362 {

364 m_LeftToRight.clear();
365 }

565template <typename LeftType, typename RightType> class ManyToMany
566{
567 std::unordered_map<LeftType, std::vector<RightType> *> m_LeftToRight;
568 std::unordered_map<RightType, std::vector<LeftType> *> m_RightToLeft;
569 std::vector<RightType> m_EmptyRightVector;
570 std::vector<LeftType> m_EmptyLeftVector;
571 int m_Count;
572
573public:
577 struct Pair
578 {
579 LeftType left;
580 RightType right;
581 Pair(LeftType left, RightType right) : left(left), right(right)
582 {
583 }
584 Pair()
585 {
586 }
587 };
588
592 ManyToMany() noexcept
593 : m_Count(0)
594 {}
595
600 void insert(const Pair &pair) noexcept
601 {
602 insert(pair.left, pair.right);
603 }
604
610 void insert(const LeftType &left, const RightType &right) noexcept
611 {
612 auto l2r_it = m_LeftToRight.find(left);
613 if (l2r_it != m_LeftToRight.end())
614 {
615 auto l2r_vec = l2r_it->second;
616 if (0 != countInSortedVector(l2r_vec, right))
617 return; // We already have this pair;
618 }
619
620 auto l2r_vec = m_LeftToRight[left]; // Will insert if it isn't already there.
621 if (l2r_vec == nullptr)
622 {
623 l2r_vec = new std::vector<RightType>();
624 m_LeftToRight[left] = l2r_vec;
625 }
626 insertIntoSortedVector(l2r_vec, right);
627
628 auto r2l_vec = m_RightToLeft[right]; // Will insert if it isn't already there.
629 if (r2l_vec == nullptr)
630 {
631 r2l_vec = new std::vector<LeftType>();
632 m_RightToLeft[right] = r2l_vec;
633 }
634 insertIntoSortedVector(r2l_vec, left);
635 m_Count += 1;
636 }
637
642 void insert(const ManyToMany<LeftType, RightType>& other) noexcept
643 {
644 for (auto pair : other)
645 {
646 insert(pair);
647 }
648 }

655 void remove(const Pair &pair) noexcept
656 {
657 remove(pair.left, pair.right);
658 }
659
666 void remove(const LeftType &left, const RightType &right) noexcept
667 {
668 auto l2r_it = m_LeftToRight.find(left);
669 if (l2r_it != m_LeftToRight.end())
670 {
671 auto r2l_it = m_RightToLeft.find(right);
672 if (r2l_it != m_RightToLeft.end())
673 {
674 auto l2r_vec = l2r_it->second;
675 auto r2l_vec = r2l_it->second;
676
677 removeFromSortedVector(l2r_vec, right);
678 removeFromSortedVector(r2l_vec, left);
679
680 if(l2r_vec->size() == 0)
681 {
682 m_LeftToRight.erase(l2r_it);
683 delete l2r_vec;
684 }
685
686 if(r2l_vec->size() == 0)
687 {
688 m_RightToLeft.erase(r2l_it);
689 delete r2l_vec;
690 }
691
692 m_Count -= 1;
693 }
694 }
695 }
696
702 void removeLeft(const LeftType &left) noexcept
703 {
704 auto l2r_it = m_LeftToRight.constFind(left);
705 if (l2r_it != m_LeftToRight.end())
706 {
707 auto l2r_vec = l2r_it->second;
708 for (auto right : l2r_vec)
709 {
710 auto r2l_it = m_RightToLeft.constFind(right);
711 auto r2l_vec = r2l_it->second;
712 removeFromSortedVector(*r2l_vec, left);
713 m_Count -= 1;
714 if (r2l_vec->size() == 0)
715 {
716 m_RightToLeft.erase(r2l_it);
717 delete r2l_vec;
718 }
719 }
720 m_LeftToRight.erase(l2r_it);
721 delete l2r_vec;
722 }
723 }

730 void removeRight(const RightType &right) noexcept
731 {
732 auto r2l_it = m_RightToLeft.constFind(right);
733 if (r2l_it != m_RightToLeft.end())
734 {
735 auto r2l_vec = r2l_it->second;
736 for (auto left : r2l_vec)
737 {
738 auto l2r_it = m_LeftToRight.constFind(left);
739 auto l2r_vec = l2r_it->second;
740 removeFromSortedVector(*l2r_vec, right);
741 m_Count -= 1;
742 if (l2r_vec->size() == 0)
743 {
744 m_LeftToRight.erase(l2r_it);
745 delete l2r_vec;
746 }
747 }
748 m_RightToLeft.erase(r2l_it);
749 delete r2l_vec;
750 }
751 }

989template <typename LeftType, typename RightType> class OneToOne
990{

998 struct Pair

1031 void insert(const LeftType &left, const RightType &right) noexcept
1032 {
1033 if (!contains(left, right))
1034 {
1035 removeLeft(left);
1036 removeRight(right);
1037 m_RightToLeft[right] = left;
1038 m_LeftToRight[left] = right;
1039 }

1071 void remove(const LeftType &left, const RightType &right) noexcept
1072 {
1073 if (contains(left, right))
1074 {
1075 auto l2r_it = m_LeftToRight.find(left);
1076 auto r2l_it = m_RightToLeft.find(right);
1077
1078 m_LeftToRight.erase(l2r_it);
1079 m_RightToLeft.erase(r2l_it);
1080 }

1089 {
1090 auto l2r_it = m_LeftToRight.find(left);
1091 if (l2r_it != m_LeftToRight.end())
1092 {
1093 auto right = l2r_it->second;
1094 auto r2l_it = m_RightToLeft.find(right);
1095 m_RightToLeft.erase(r2l_it);
1096 m_LeftToRight.erase(l2r_it);
