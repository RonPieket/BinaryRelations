37// -------- Manipulate vector with unique sorted elements --------
38
39template <typename T> int countInSortedVector(const std::vector<T> *vector, const T &value) noexcept
40{
41 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
42 return it != vector->cend() && *it == value ? 1 : 0;
43}
44
45template <typename T>
46typename std::vector<T>::const_iterator findInSortedVector(const std::vector<T> *vector, const T &value) noexcept
47{
48 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
49 if (it != vector->cend() && *it == value)
50 return it;
51 else
52 return vector->cend();
53}
54
55template <typename T> int insertIntoSortedVector(std::vector<T> *vector, const T &value) noexcept
56{
57 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
58 if (it != vector->end() && *it == value)
59 return 0; // It's already there
60 else
61 {
62 vector->insert(it, value);
63 return 1;
64 }
65}
66
67template <typename T> int removeFromSortedVector(std::vector<T> *vector, const T &value) noexcept
68{
69 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
70 if (it != vector->end() && *it == value)
71 {
72 vector->erase(it);
73 return 1;
74 }
75 else
76 return 0;
77}

79template <typename T>
80void insertIntoSortedVector(std::vector<T> *sourceVector, std::vector<T> *insertVector, std::vector<T> *outVector) noexcept
81{
82 typename std::vector<T>::const_iterator sourceIt = sourceVector->cbegin();
83 typename std::vector<T>::const_iterator insertIt = insertVector->cbegin();
84
85 outVector->clear();
86 outVector->reserve(sourceVector->count() + insertVector->count());
87
88 while (sourceIt != sourceVector->cend() && insertIt != insertVector->cend())
89 {
90 if (*sourceIt == *insertIt)
91 {
92 outVector->append(*sourceIt++);
93 insertIt++;
94 }
95 else if (*sourceIt < *insertIt)
96 outVector->append(*sourceIt++);
97 else
98 outVector->append(*insertIt++);
99 }
100
101 while (sourceIt != sourceVector->cend())
102 outVector->append(*sourceIt++);
103
104 while (insertIt != insertVector->cend())
105 outVector->append(*insertIt++);
106}
107
108template <typename T>
109void removeFromSortedVector(std::vector<T> *sourceVector, std::vector<T> *removeVector, std::vector<T> *outVector) noexcept
110{
111 typename std::vector<T>::const_iterator sourceIt = sourceVector->cbegin();
112 typename std::vector<T>::const_iterator removeIt = removeVector->cbegin();
113
114 outVector->clear();
115 outVector->reserve(std::max(sourceVector->count(), removeVector->count()));
116
117 while (sourceIt != sourceVector->cend() && removeIt != removeVector->cend())
118 {
119 if (*sourceIt == *removeIt)
120 {
121 sourceIt++;
122 removeIt++;
123 }
124 else if (*sourceIt < *removeIt)
125 outVector->append(*sourceIt++);
126 else
127 outVector->append(*removeIt++);
128 }
129
130 while (sourceIt != sourceVector->cend())
131 outVector->append(*sourceIt++);
132}

140template <typename LeftType, typename RightType> class OneToMany
141{
142 std::unordered_map<LeftType, std::vector<RightType> *> m_LeftToRight;
143 std::unordered_map<RightType, LeftType> m_RightToLeft;
144 std::vector<RightType> m_EmptyRightVector;
145
146public:
150 struct Pair
151 {
152 LeftType left;
153 RightType right;
154 Pair(LeftType left, RightType right)
155 : left(left), right(right)
156 {}
157 Pair()
158 {}
159 };

164 OneToMany() noexcept
165 {}

172 void insert(const Pair &pair) noexcept

183 void insert(const LeftType &left, const RightType &right) noexcept
184 {
185 auto r2l_it = m_RightToLeft.find(right);
186 if (r2l_it != m_RightToLeft.end())
187 {
188 if (r2l_it->second == left)
189 return; // We already have this pair - do nothing
190
191 remove(r2l_it->second, right); // Remove old relation
192 }
193
194 auto l2r_vec = m_LeftToRight[left]; // Will insert if it isn't already there.
195 if (l2r_vec == nullptr)
196 {
197 l2r_vec = new std::vector<RightType>();
198 m_LeftToRight[left] = l2r_vec;
199 }
200 insertIntoSortedVector(l2r_vec, right);
201
202 m_RightToLeft[right] = left;
203 }
204
210 void insert(const OneToMany<LeftType, RightType>& other) noexcept
211 {
212 for (auto pair : other)
213 {
214 insert(pair);
215 }
216 }
217
223 void remove(const Pair &pair) noexcept
224 {
225 remove(pair.left, pair.right);
226 }
227
234 void remove(const LeftType &left, const RightType &right) noexcept
235 {
236 auto r2l_it = m_RightToLeft.find(right);
237 if (r2l_it == m_RightToLeft.end())
238 return; // (*,right) not in the set - do nothing
239
240 if (r2l_it->second != left)
241 return; // (left,right) is not in the set - do nothing
242
243 auto l2r_it = m_LeftToRight.find(left);
244 auto l2r_vec = l2r_it->second;
245 auto l2r_vec_it = findInSortedVector(l2r_vec, right);
246 if (l2r_vec_it != l2r_vec->cend())
247 {
248 l2r_vec->erase(l2r_vec_it);
249 if (l2r_vec->size() == 0)
250 {
251 m_LeftToRight.erase(l2r_it);
252 delete l2r_vec; // Vector is empty now
253 }
254 m_RightToLeft.erase(r2l_it);
255 }
256 }
257
263 void removeLeft(const LeftType &left) noexcept
264 {
265 auto l2r_it = m_LeftToRight.find(left);
266 if (l2r_it == m_LeftToRight.end())
267 return;
268
269 auto l2r_vec = l2r_it->second;
270 for (auto right : l2r_vec)
271 {
272 auto r2l_it = m_RightToLeft.find(right);
273 m_RightToLeft.erase(r2l_it);
274 }
275
276 m_LeftToRight.erase(l2r_it);
277 delete l2r_vec; // Vector is empty now
278 }
279
285 void removeRight(const RightType &right) noexcept
286 {
287 auto r2l_it = m_RightToLeft.find(right);
288 if (r2l_it == m_RightToLeft.end())
289 return;
290 remove(r2l_it->second, right);
291 }
292
297 void remove(const OneToMany<LeftType, RightType> &other) noexcept
298 {
299 for (auto pair : other)
300 {
301 remove(pair);
302 }
303 }
304
308 void clear() noexcept
309 {
310 m_RightToLeft.clear();
311 m_LeftToRight.clear();
312 }
313
318 bool contains(const Pair &pair) const noexcept
319 {
320 return contains(pair.left, pair.right);
321 }
322
328 bool contains(const LeftType &left, const RightType &right) const noexcept
329 {
330 auto r2l_it = m_RightToLeft.find(right);
331 return r2l_it != m_RightToLeft.cend() && r2l_it->second == left;
332 }

341 }
342
347 bool containsRight(const RightType &right) const noexcept

350 }

362 if (l2r_it == m_LeftToRight.end())
363 return &m_EmptyRightVector;

365 return l2r_it->second;
366 }

378 if (r2l_it == m_RightToLeft.end())
379 return notFoundValue;

381 return r2l_it->second;

414 class Iterator
415 {
419 public:
420 typename std::unordered_map<LeftType, std::vector<RightType> *>::const_iterator l2r_it;
421 typename std::unordered_map<LeftType, std::vector<RightType> *>::const_iterator l2r_it_end;
422 typename std::vector<RightType>::const_iterator l2r_vec_it;
423
424 inline Pair operator*() const noexcept
425 {
426 auto left = l2r_it->first;
427 auto right = *l2r_vec_it;
428 return Pair(left, right);
429 }
430
431 inline bool operator==(const Iterator &other) const noexcept
432 {
433 return l2r_it == other.l2r_it;
434 }
435
436 inline bool operator!=(const Iterator &other) const noexcept
437 {
438 return l2r_it != other.l2r_it;
439 }
440
441 inline Iterator operator++() noexcept
442 {
443 l2r_vec_it++;
444 if (l2r_vec_it == l2r_it->second->cend())
445 {
446 l2r_it++;
447 if (l2r_it != l2r_it_end)
448 {
449 l2r_vec_it = l2r_it->second->cbegin();
450 }
451 }
452 return *this;
453 }
457 };
458
463 Iterator begin() const noexcept
464 {
465 Iterator it;
466 it.l2r_it = m_LeftToRight.cbegin();
467 it.l2r_it_end = m_LeftToRight.cend();
468 if (it.l2r_it != it.l2r_it_end)
469 {
470 auto l2r_vec = it.l2r_it->second;
471 it.l2r_vec_it = l2r_vec->cbegin();
472 }
473 return it;
474 }
475
480 Iterator end() const noexcept
481 {
482 Iterator it;
483 it.l2r_it = m_LeftToRight.cend();
484 return it;
485 }

494template <typename LeftType, typename RightType> class ManyToMany
495{
496 std::unordered_map<LeftType, std::vector<RightType> *> m_LeftToRight;
497 std::unordered_map<RightType, std::vector<LeftType> *> m_RightToLeft;
498 std::vector<RightType> m_EmptyRightVector;
499 std::vector<LeftType> m_EmptyLeftVector;
500 int m_Count;
501
502public:
506 struct Pair
507 {
508 LeftType left;
509 RightType right;
510 Pair(LeftType left, RightType right) : left(left), right(right)
511 {
512 }
513 Pair()
514 {
515 }
516 };
517
521 ManyToMany() noexcept
522 : m_Count(0)
523 {}
524
529 void insert(const Pair &pair) noexcept
530 {
531 insert(pair.left, pair.right);
532 }
533
539 void insert(const LeftType &left, const RightType &right) noexcept
540 {
541 auto l2r_it = m_LeftToRight.find(left);
542 if (l2r_it != m_LeftToRight.end())
543 {
544 auto l2r_vec = l2r_it->second;
545 if (0 != countInSortedVector(l2r_vec, right))
546 return; // We already have this pair;
547 }
548
549 auto l2r_vec = m_LeftToRight[left]; // Will insert if it isn't already there.
550 if (l2r_vec == nullptr)
551 {
552 l2r_vec = new std::vector<RightType>();
553 m_LeftToRight[left] = l2r_vec;
554 }
555 insertIntoSortedVector(l2r_vec, right);
556
557 auto r2l_vec = m_RightToLeft[right]; // Will insert if it isn't already there.
558 if (r2l_vec == nullptr)
559 {
560 r2l_vec = new std::vector<LeftType>();
561 m_RightToLeft[right] = r2l_vec;
562 }
563 insertIntoSortedVector(r2l_vec, left);
564 m_Count += 1;
565 }
566
571 void insert(const ManyToMany<LeftType, RightType>& other) noexcept
572 {
573 for (auto pair : other)
574 {
575 insert(pair);
576 }
577 }
578
584 void remove(const Pair &pair) noexcept
585 {
586 remove(pair.left, pair.right);
587 }
588
595 void remove(const LeftType &left, const RightType &right) noexcept
596 {
597 auto l2r_it = m_LeftToRight.find(left);
598 if (l2r_it != m_LeftToRight.end())
599 {
600 auto r2l_it = m_RightToLeft.find(right);
601 if (r2l_it != m_RightToLeft.end())
602 {
603 auto l2r_vec = l2r_it->second;
604 auto r2l_vec = r2l_it->second;
605
606 removeFromSortedVector(l2r_vec, right);
607 removeFromSortedVector(r2l_vec, left);
608
609 if(l2r_vec->size() == 0)
610 {
611 m_LeftToRight.erase(l2r_it);
612 delete l2r_vec;
613 }
614
615 if(r2l_vec->size() == 0)
616 {
617 m_RightToLeft.erase(r2l_it);
618 delete r2l_vec;
619 }
620
621 m_Count -= 1;
622 }
623 }
624 }
625
631 void removeLeft(const LeftType &left) noexcept
632 {
633 auto l2r_it = m_LeftToRight.constFind(left);
634 if (l2r_it != m_LeftToRight.end())
635 {
636 auto l2r_vec = l2r_it->second;
637 for (auto right : l2r_vec)
638 {
639 auto r2l_it = m_RightToLeft.constFind(right);
640 auto r2l_vec = r2l_it->second;
641 removeFromSortedVector(*r2l_vec, left);
642 m_Count -= 1;
643 if (r2l_vec->size() == 0)
644 {
645 m_RightToLeft.erase(r2l_it);
646 delete r2l_vec;
647 }
648 }
649 m_LeftToRight.erase(l2r_it);
650 delete l2r_vec;
651 }
652 }
653
659 void removeRight(const RightType &right) noexcept
660 {
661 auto r2l_it = m_RightToLeft.constFind(right);
662 if (r2l_it != m_RightToLeft.end())
663 {
664 auto r2l_vec = r2l_it->second;
665 for (auto left : r2l_vec)
666 {
667 auto l2r_it = m_LeftToRight.constFind(left);
668 auto l2r_vec = l2r_it->second;
669 removeFromSortedVector(*l2r_vec, right);
670 m_Count -= 1;
671 if (l2r_vec->size() == 0)
672 {
673 m_LeftToRight.erase(l2r_it);
674 delete l2r_vec;
675 }
676 }
677 m_RightToLeft.erase(r2l_it);
678 delete r2l_vec;
679 }
680 }

720 if (l2r_it != m_LeftToRight.end())
721 {
722 auto r2l_it = m_RightToLeft.find(right);

734 }
735 }
736 }
737 return false;
738 }
