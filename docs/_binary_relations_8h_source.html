37// -------- Manipulate vector with unique sorted elements --------
38
39template <typename T> int countInSortedVector(const std::vector<T> *vector, const T &value) noexcept
40{
41 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
42 return it != vector->cend() && *it == value ? 1 : 0;
43}
44
45template <typename T>
46typename std::vector<T>::const_iterator findInSortedVector(const std::vector<T> *vector, const T &value) noexcept
47{
48 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
49 if (it != vector->cend() && *it == value)
50 return it;
51 else
52 return vector->cend();
53}
54
55template <typename T> int insertIntoSortedVector(std::vector<T> *vector, const T &value) noexcept
56{
57 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
58 if (it != vector->end() && *it == value)
59 return 0; // It's already there
60 else
61 {
62 vector->insert(it, value);
63 return 1;
64 }
65}
66
67template <typename T> int eraseFromSortedVector(std::vector<T> *vector, const T &value) noexcept
68{
69 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
70 if (it != vector->end() && *it == value)
71 {
72 vector->erase(it);
73 return 1;
74 }
75 else
76 return 0;
77}
78
79template <typename T>
80void insertIntoSortedVector(std::vector<T> *sourceVector, std::vector<T> *insertVector, std::vector<T> *outVector) noexcept
81{
82 typename std::vector<T>::const_iterator sourceIt = sourceVector->cbegin();
83 typename std::vector<T>::const_iterator insertIt = insertVector->cbegin();
84
85 outVector->clear();
86 outVector->reserve(sourceVector->count() + insertVector->count());
87
88 while (sourceIt != sourceVector->cend() && insertIt != insertVector->cend())
89 {
90 if (*sourceIt == *insertIt)
91 {
92 outVector->append(*sourceIt++);
93 insertIt++;
94 }
95 else if (*sourceIt < *insertIt)
96 outVector->append(*sourceIt++);
97 else
98 outVector->append(*insertIt++);
99 }
100
101 while (sourceIt != sourceVector->cend())
102 outVector->append(*sourceIt++);
103
104 while (insertIt != insertVector->cend())
105 outVector->append(*insertIt++);
106}
107
108template <typename T>
109void eraseFromSortedVector(std::vector<T> *sourceVector, std::vector<T> *eraseVector, std::vector<T> *outVector) noexcept
110{
111 typename std::vector<T>::const_iterator sourceIt = sourceVector->cbegin();
112 typename std::vector<T>::const_iterator eraseIt = eraseVector->cbegin();
113
114 outVector->clear();
115 outVector->reserve(std::max(sourceVector->count(), eraseVector->count()));
116
117 while (sourceIt != sourceVector->cend() && eraseIt != eraseVector->cend())
118 {
119 if (*sourceIt == *eraseIt)
120 {
121 sourceIt++;
122 eraseIt++;
123 }
124 else if (*sourceIt < *eraseIt)
125 outVector->append(*sourceIt++);
126 else
127 outVector->append(*eraseIt++);
128 }
129
130 while (sourceIt != sourceVector->cend())
131 outVector->append(*sourceIt++);
132}

195template <typename LeftType, typename RightType> class OneToMany
196{
197 std::unordered_map<LeftType, std::vector<RightType> *> m_LeftToRight;
198 std::unordered_map<RightType, LeftType> m_RightToLeft;
199 std::vector<RightType> m_EmptyRightVector;
200 bool m_RightIsSorted = false;
201
202public:
206 struct Pair
207 {
208 LeftType left;
209 RightType right;
210 Pair(LeftType left, RightType right)
211 : left(left), right(right)
212 {}
213 Pair()
214 {}
215 };
216
220 OneToMany() noexcept
221 {}

228 void insert(const Pair &pair) noexcept
229 {
230 insert(pair.left, pair.right);
231 }

239 void insert(const LeftType &left, const RightType &right) noexcept
240 {
241 auto r2l_it = m_RightToLeft.find(right);
242 if (r2l_it != m_RightToLeft.end())
243 {
244 if (r2l_it->second == left)
245 return; // We already have this pair - do nothing
246
247 erase(r2l_it->second, right); // Erase old relation
248 }
249
250 auto l2r_vec = m_LeftToRight[left]; // Will insert if it isn't already there.
251 if (l2r_vec == nullptr)
252 {
253 l2r_vec = new std::vector<RightType>();
254 m_LeftToRight[left] = l2r_vec;
255 }
256 insertIntoSortedVector(l2r_vec, right);
257
258 m_RightToLeft[right] = left;
259 }

266 void insert(const OneToMany<LeftType, RightType>& other) noexcept
267 {
268 for (auto pair : other)
269 {
270 insert(pair);
271 }
272 }

279 void erase(const Pair &pair) noexcept
280 {
281 erase(pair.left, pair.right);
282 }

290 void erase(const LeftType &left, const RightType &right) noexcept
291 {
292 auto r2l_it = m_RightToLeft.find(right);
293 if (r2l_it == m_RightToLeft.end())
294 return; // (*,right) not in the set - do nothing
295
296 if (r2l_it->second != left)
297 return; // (left,right) is not in the set - do nothing
298
299 auto l2r_it = m_LeftToRight.find(left);
300 auto l2r_vec = l2r_it->second;
301 auto l2r_vec_it = findInSortedVector(l2r_vec, right);
302 if (l2r_vec_it != l2r_vec->cend())
303 {
304 l2r_vec->erase(l2r_vec_it);
305 if (l2r_vec->size() == 0)
306 {
307 m_LeftToRight.erase(l2r_it);
308 delete l2r_vec; // Vector is empty now
309 }
310 m_RightToLeft.erase(r2l_it);
311 }
312 }
313
319 void eraseLeft(const LeftType &left) noexcept
320 {
321 auto l2r_it = m_LeftToRight.find(left);
322 if (l2r_it == m_LeftToRight.end())
323 return;
324
325 auto l2r_vec = l2r_it->second;
326 for (auto right : l2r_vec)
327 {
328 auto r2l_it = m_RightToLeft.find(right);
329 m_RightToLeft.erase(r2l_it);
330 }
331
332 m_LeftToRight.erase(l2r_it);
333 delete l2r_vec; // Vector is empty now
334 }

341 void eraseRight(const RightType &right) noexcept
342 {
343 auto r2l_it = m_RightToLeft.find(right);
344 if (r2l_it == m_RightToLeft.end())
345 return;
346 erase(r2l_it->second, right);
347 }

355 for (auto pair : other)
356 {
357 erase(pair);
358 }
359 }

488 class Iterator
489 {
491 public:
492 typename std::unordered_map<LeftType, std::vector<RightType> *>::const_iterator l2r_it;
493 typename std::unordered_map<LeftType, std::vector<RightType> *>::const_iterator l2r_it_end;
494 typename std::vector<RightType>::const_iterator l2r_vec_it;
495
496 inline Pair operator*() const noexcept
497 {
498 auto left = l2r_it->first;
499 auto right = *l2r_vec_it;
500 return Pair(left, right);
501 }
502
503 inline bool operator==(const Iterator &other) const noexcept
504 {
505 return l2r_it == other.l2r_it;
506 }
507
508 inline bool operator!=(const Iterator &other) const noexcept
509 {
510 return l2r_it != other.l2r_it;
511 }
512
513 inline Iterator operator++() noexcept
514 {
515 l2r_vec_it++;
516 if (l2r_vec_it == l2r_it->second->cend())
517 {
518 l2r_it++;
519 if (l2r_it != l2r_it_end)
520 {
521 l2r_vec_it = l2r_it->second->cbegin();
522 }
523 }
524 return *this;
525 }
527 };
528
533 Iterator begin() const noexcept
534 {
535 Iterator it;
536 it.l2r_it = m_LeftToRight.cbegin();
537 it.l2r_it_end = m_LeftToRight.cend();
538 if (it.l2r_it != it.l2r_it_end)
539 {
540 auto l2r_vec = it.l2r_it->second;
541 it.l2r_vec_it = l2r_vec->cbegin();
542 }
543 return it;
544 }
545
550 Iterator end() const noexcept
551 {
552 Iterator it;
553 it.l2r_it = m_LeftToRight.cend();
554 return it;
555 }
556};

564template <typename LeftType, typename RightType> class ManyToMany
565{
566 std::unordered_map<LeftType, std::vector<RightType> *> m_LeftToRight;
567 std::unordered_map<RightType, std::vector<LeftType> *> m_RightToLeft;
568 std::vector<RightType> m_EmptyRightVector;
569 std::vector<LeftType> m_EmptyLeftVector;
570 int m_Count;
571
572public:
576 struct Pair
577 {
578 LeftType left;
579 RightType right;
580 Pair(LeftType left, RightType right) : left(left), right(right)
581 {
582 }
583 Pair()
584 {
585 }
586 };
587
591 ManyToMany() noexcept
592 : m_Count(0)
593 {}
594
599 void insert(const Pair &pair) noexcept
600 {
601 insert(pair.left, pair.right);
602 }
603
609 void insert(const LeftType &left, const RightType &right) noexcept
610 {
611 auto l2r_it = m_LeftToRight.find(left);
612 if (l2r_it != m_LeftToRight.end())
613 {
614 auto l2r_vec = l2r_it->second;
615 if (0 != countInSortedVector(l2r_vec, right))
616 return; // We already have this pair;
617 }
618
619 auto l2r_vec = m_LeftToRight[left]; // Will insert if it isn't already there.
620 if (l2r_vec == nullptr)
621 {
622 l2r_vec = new std::vector<RightType>();
623 m_LeftToRight[left] = l2r_vec;
624 }
625 insertIntoSortedVector(l2r_vec, right);
626
627 auto r2l_vec = m_RightToLeft[right]; // Will insert if it isn't already there.
628 if (r2l_vec == nullptr)
629 {
630 r2l_vec = new std::vector<LeftType>();
631 m_RightToLeft[right] = r2l_vec;
632 }
633 insertIntoSortedVector(r2l_vec, left);
634 m_Count += 1;
635 }
636
641 void insert(const ManyToMany<LeftType, RightType>& other) noexcept
642 {
643 for (auto pair : other)
644 {
645 insert(pair);
646 }
647 }

654 void erase(const Pair &pair) noexcept
655 {
656 erase(pair.left, pair.right);
657 }
658
665 void erase(const LeftType &left, const RightType &right) noexcept
666 {
667 auto l2r_it = m_LeftToRight.find(left);
668 if (l2r_it != m_LeftToRight.end())
669 {
670 auto r2l_it = m_RightToLeft.find(right);
671 if (r2l_it != m_RightToLeft.end())
672 {
673 auto l2r_vec = l2r_it->second;
674 auto r2l_vec = r2l_it->second;
675
676 eraseFromSortedVector(l2r_vec, right);
677 eraseFromSortedVector(r2l_vec, left);
678
679 if(l2r_vec->size() == 0)
680 {
681 m_LeftToRight.erase(l2r_it);
682 delete l2r_vec;
683 }
684
685 if(r2l_vec->size() == 0)
686 {
687 m_RightToLeft.erase(r2l_it);
688 delete r2l_vec;
689 }
690
691 m_Count -= 1;
692 }
693 }
694 }
695
701 void eraseLeft(const LeftType &left) noexcept
702 {
703 auto l2r_it = m_LeftToRight.constFind(left);
704 if (l2r_it != m_LeftToRight.end())
705 {
706 auto l2r_vec = l2r_it->second;
707 for (auto right : l2r_vec)
708 {
709 auto r2l_it = m_RightToLeft.constFind(right);
710 auto r2l_vec = r2l_it->second;
711 eraseFromSortedVector(*r2l_vec, left);
712 m_Count -= 1;
713 if (r2l_vec->size() == 0)
714 {
715 m_RightToLeft.erase(r2l_it);
716 delete r2l_vec;
717 }
718 }
719 m_LeftToRight.erase(l2r_it);
720 delete l2r_vec;
721 }
722 }
723
729 void eraseRight(const RightType &right) noexcept
730 {
731 auto r2l_it = m_RightToLeft.constFind(right);
732 if (r2l_it != m_RightToLeft.end())
733 {
734 auto r2l_vec = r2l_it->second;
735 for (auto left : r2l_vec)
736 {
737 auto l2r_it = m_LeftToRight.constFind(left);
738 auto l2r_vec = l2r_it->second;
739 eraseFromSortedVector(*l2r_vec, right);
740 m_Count -= 1;
741 if (l2r_vec->size() == 0)
742 {
743 m_LeftToRight.erase(l2r_it);
744 delete l2r_vec;
745 }
746 }
747 m_RightToLeft.erase(r2l_it);
748 delete r2l_vec;
749 }
750 }

787 bool contains(const LeftType &left, const RightType &right) const noexcept
788 {

791 {

804 }
805 }
806 }
807 return false;
808 }
