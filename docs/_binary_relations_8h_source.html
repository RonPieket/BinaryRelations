37// -------- Manipulate vector with unique sorted elements --------
38
39template <typename T> int countInSortedVector(const std::vector<T> *vector, const T &value) noexcept
40{
41 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
42 return it != vector->cend() && *it == value ? 1 : 0;
43}
44
45template <typename T>
46typename std::vector<T>::const_iterator findInSortedVector(const std::vector<T> *vector, const T &value) noexcept
47{
48 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
49 if (it != vector->cend() && *it == value)
50 return it;
51 else
52 return vector->cend();
53}
54
55template <typename T> int appendToSortedVector(std::vector<T> *vector, const T &value) noexcept
56{
57 vector->append(value);
58 return 1;
59}
60
61template <typename T> int insertIntoSortedVector(std::vector<T> *vector, const T &value) noexcept
62{
63 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
64 if (it != vector->end() && *it == value)
65 return 0; // It's already there
66 else
67 {
68 vector->insert(it, value);
69 return 1;
70 }
71}
72
73template <typename T> int removeFromSortedVector(std::vector<T> *vector, const T &value) noexcept
74{
75 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
76 if (it != vector->end() && *it == value)
77 {
78 vector->erase(it);
79 return 1;
80 }
81 else
82 return 0;
83}
84
85template <typename T>
86void insertIntoSortedVector(std::vector<T> *sourceVector, std::vector<T> *insertVector, std::vector<T> *outVector) noexcept
87{
88 typename std::vector<T>::const_iterator sourceIt = sourceVector->cbegin();
89 typename std::vector<T>::const_iterator insertIt = insertVector->cbegin();
90
91 outVector->clear();
92 outVector->reserve(sourceVector->count() + insertVector->count());
93
94 while (sourceIt != sourceVector->cend() && insertIt != insertVector->cend())
95 {
96 if (*sourceIt == *insertIt)
97 {
98 outVector->append(*sourceIt++);
99 insertIt++;
100 }
101 else if (*sourceIt < *insertIt)
102 outVector->append(*sourceIt++);
103 else
104 outVector->append(*insertIt++);
105 }
106
107 while (sourceIt != sourceVector->cend())
108 outVector->append(*sourceIt++);
109
110 while (insertIt != insertVector->cend())
111 outVector->append(*insertIt++);
112}
113
114template <typename T>
115void removeFromSortedVector(std::vector<T> *sourceVector, std::vector<T> *removeVector, std::vector<T> *outVector) noexcept
116{
117 typename std::vector<T>::const_iterator sourceIt = sourceVector->cbegin();
118 typename std::vector<T>::const_iterator removeIt = removeVector->cbegin();
119
120 outVector->clear();
121 outVector->reserve(std::max(sourceVector->count(), removeVector->count()));
122
123 while (sourceIt != sourceVector->cend() && removeIt != removeVector->cend())
124 {
125 if (*sourceIt == *removeIt)
126 {
127 sourceIt++;
128 removeIt++;
129 }
130 else if (*sourceIt < *removeIt)
131 outVector->append(*sourceIt++);
132 else
133 outVector->append(*removeIt++);
134 }
135
136 while (sourceIt != sourceVector->cend())
137 outVector->append(*sourceIt++);
138}

201template <typename LeftType, typename RightType> class OneToMany
202{
203 std::unordered_map<LeftType, std::vector<RightType> *> m_LeftToRight;
204 std::unordered_map<RightType, LeftType> m_RightToLeft;
205 std::vector<RightType> m_EmptyRightVector;
206 bool m_RightIsSorted = false;
207
208public:
212 struct Pair
213 {
214 LeftType left;
215 RightType right;
216 Pair(LeftType left, RightType right)
217 : left(left), right(right)
218 {}
219 Pair()
220 {}
221 };
222
226 OneToMany() noexcept
227 {}

234 void insert(const Pair &pair) noexcept
235 {
236 insert(pair.left, pair.right);

245 void insert(const LeftType &left, const RightType &right) noexcept
246 {
247 auto r2l_it = m_RightToLeft.find(right);
248 if (r2l_it != m_RightToLeft.end())
249 {
250 if (r2l_it->second == left)
251 return; // We already have this pair - do nothing
252
253 remove(r2l_it->second, right); // Remove old relation
254 }
255
256 auto l2r_vec = m_LeftToRight[left]; // Will insert if it isn't already there.
257 if (l2r_vec == nullptr)
258 {
259 l2r_vec = new std::vector<RightType>();
260 m_LeftToRight[left] = l2r_vec;
261 }
262 insertIntoSortedVector(l2r_vec, right);
263
264 m_RightToLeft[right] = left;
265 }
266
272 void insert(const OneToMany<LeftType, RightType>& other) noexcept
273 {
274 for (auto pair : other)
275 {
276 insert(pair);
277 }
278 }
279
285 void remove(const Pair &pair) noexcept
286 {
287 remove(pair.left, pair.right);
288 }
289
296 void remove(const LeftType &left, const RightType &right) noexcept
297 {
298 auto r2l_it = m_RightToLeft.find(right);
299 if (r2l_it == m_RightToLeft.end())
300 return; // (*,right) not in the set - do nothing
301
302 if (r2l_it->second != left)
303 return; // (left,right) is not in the set - do nothing
304
305 auto l2r_it = m_LeftToRight.find(left);
306 auto l2r_vec = l2r_it->second;
307 auto l2r_vec_it = findInSortedVector(l2r_vec, right);
308 if (l2r_vec_it != l2r_vec->cend())
309 {
310 l2r_vec->erase(l2r_vec_it);
311 if (l2r_vec->size() == 0)
312 {
313 m_LeftToRight.erase(l2r_it);
314 delete l2r_vec; // Vector is empty now
315 }
316 m_RightToLeft.erase(r2l_it);
317 }
318 }
319
325 void removeLeft(const LeftType &left) noexcept
326 {
327 auto l2r_it = m_LeftToRight.find(left);
328 if (l2r_it == m_LeftToRight.end())
329 return;
330
331 auto l2r_vec = l2r_it->second;
332 for (auto right : l2r_vec)
333 {
334 auto r2l_it = m_RightToLeft.find(right);
335 m_RightToLeft.erase(r2l_it);
336 }
337
338 m_LeftToRight.erase(l2r_it);
339 delete l2r_vec; // Vector is empty now
340 }
341
347 void removeRight(const RightType &right) noexcept
348 {
349 auto r2l_it = m_RightToLeft.find(right);
350 if (r2l_it == m_RightToLeft.end())
351 return;
352 remove(r2l_it->second, right);
353 }
354
359 void remove(const OneToMany<LeftType, RightType> &other) noexcept
360 {
361 for (auto pair : other)
362 {
363 remove(pair);
364 }
365 }
366
370 void clear() noexcept
371 {
372 m_RightToLeft.clear();
373 m_LeftToRight.clear();
374 }

381 {
382 return contains(pair.left, pair.right);
383 }
384

391 {
392 auto r2l_it = m_RightToLeft.find(right);
393 return r2l_it != m_RightToLeft.cend() && r2l_it->second == left;

498 typename std::unordered_map<LeftType, std::vector<RightType> *>::const_iterator l2r_it;
499 typename std::unordered_map<LeftType, std::vector<RightType> *>::const_iterator l2r_it_end;
500 typename std::vector<RightType>::const_iterator l2r_vec_it;
501

505 auto right = *l2r_vec_it;
506 return Pair(left, right);
507 }
508
509 inline bool operator==(const Iterator &other) const noexcept
510 {
511 return l2r_it == other.l2r_it;
512 }
513
514 inline bool operator!=(const Iterator &other) const noexcept
515 {
516 return l2r_it != other.l2r_it;
517 }
518
519 inline Iterator operator++() noexcept
520 {
521 l2r_vec_it++;
522 if (l2r_vec_it == l2r_it->second->cend())
523 {
524 l2r_it++;
525 if (l2r_it != l2r_it_end)
526 {
527 l2r_vec_it = l2r_it->second->cbegin();
528 }
529 }
530 return *this;
531 }
533 };
534
539 Iterator begin() const noexcept
540 {
541 Iterator it;
542 it.l2r_it = m_LeftToRight.cbegin();
543 it.l2r_it_end = m_LeftToRight.cend();
544 if (it.l2r_it != it.l2r_it_end)
545 {
546 auto l2r_vec = it.l2r_it->second;
547 it.l2r_vec_it = l2r_vec->cbegin();
548 }
549 return it;
550 }
551
556 Iterator end() const noexcept
557 {
558 Iterator it;
559 it.l2r_it = m_LeftToRight.cend();
560 return it;
561 }

570template <typename LeftType, typename RightType> class ManyToMany
571{
572 std::unordered_map<LeftType, std::vector<RightType> *> m_LeftToRight;
573 std::unordered_map<RightType, std::vector<LeftType> *> m_RightToLeft;
574 std::vector<RightType> m_EmptyRightVector;
575 std::vector<LeftType> m_EmptyLeftVector;
576 int m_Count;
577
578public:
582 struct Pair
583 {
584 LeftType left;
585 RightType right;
586 Pair(LeftType left, RightType right) : left(left), right(right)
587 {
588 }
589 Pair()
590 {
591 }
592 };
593
597 ManyToMany() noexcept
598 : m_Count(0)
599 {}
600
605 void insert(const Pair &pair) noexcept
606 {
607 insert(pair.left, pair.right);
608 }
609
615 void insert(const LeftType &left, const RightType &right) noexcept
616 {
617 auto l2r_it = m_LeftToRight.find(left);
618 if (l2r_it != m_LeftToRight.end())
619 {
620 auto l2r_vec = l2r_it->second;
621 if (0 != countInSortedVector(l2r_vec, right))
622 return; // We already have this pair;
623 }
624
625 auto l2r_vec = m_LeftToRight[left]; // Will insert if it isn't already there.
626 if (l2r_vec == nullptr)
627 {
628 l2r_vec = new std::vector<RightType>();
629 m_LeftToRight[left] = l2r_vec;
630 }
631 insertIntoSortedVector(l2r_vec, right);
632
633 auto r2l_vec = m_RightToLeft[right]; // Will insert if it isn't already there.
634 if (r2l_vec == nullptr)
635 {
636 r2l_vec = new std::vector<LeftType>();
637 m_RightToLeft[right] = r2l_vec;
638 }
639 insertIntoSortedVector(r2l_vec, left);
640 m_Count += 1;
641 }
642
647 void insert(const ManyToMany<LeftType, RightType>& other) noexcept
648 {
649 for (auto pair : other)
650 {
651 insert(pair);
652 }
653 }
654
660 void remove(const Pair &pair) noexcept
661 {
662 remove(pair.left, pair.right);
663 }
664
671 void remove(const LeftType &left, const RightType &right) noexcept
672 {
673 auto l2r_it = m_LeftToRight.find(left);
674 if (l2r_it != m_LeftToRight.end())
675 {
676 auto r2l_it = m_RightToLeft.find(right);
677 if (r2l_it != m_RightToLeft.end())
678 {
679 auto l2r_vec = l2r_it->second;
680 auto r2l_vec = r2l_it->second;
681
682 removeFromSortedVector(l2r_vec, right);
683 removeFromSortedVector(r2l_vec, left);
684
685 if(l2r_vec->size() == 0)
686 {
687 m_LeftToRight.erase(l2r_it);
688 delete l2r_vec;
689 }
690
691 if(r2l_vec->size() == 0)
692 {
693 m_RightToLeft.erase(r2l_it);
694 delete r2l_vec;
695 }
696
697 m_Count -= 1;
698 }
699 }
700 }
701
707 void removeLeft(const LeftType &left) noexcept
708 {
709 auto l2r_it = m_LeftToRight.constFind(left);
710 if (l2r_it != m_LeftToRight.end())
711 {
712 auto l2r_vec = l2r_it->second;
713 for (auto right : l2r_vec)
714 {
715 auto r2l_it = m_RightToLeft.constFind(right);
716 auto r2l_vec = r2l_it->second;
717 removeFromSortedVector(*r2l_vec, left);
718 m_Count -= 1;
719 if (r2l_vec->size() == 0)
720 {
721 m_RightToLeft.erase(r2l_it);
722 delete r2l_vec;
723 }
724 }
725 m_LeftToRight.erase(l2r_it);
726 delete l2r_vec;
727 }
728 }
729
735 void removeRight(const RightType &right) noexcept
736 {
737 auto r2l_it = m_RightToLeft.constFind(right);
738 if (r2l_it != m_RightToLeft.end())
739 {
740 auto r2l_vec = r2l_it->second;
741 for (auto left : r2l_vec)
742 {
743 auto l2r_it = m_LeftToRight.constFind(left);
744 auto l2r_vec = l2r_it->second;
745 removeFromSortedVector(*l2r_vec, right);
746 m_Count -= 1;
747 if (l2r_vec->size() == 0)
748 {
749 m_LeftToRight.erase(l2r_it);
750 delete l2r_vec;
751 }
752 }
753 m_RightToLeft.erase(r2l_it);
754 delete r2l_vec;
755 }
756 }
757
762 void remove(const ManyToMany<LeftType, RightType>& other) noexcept
763 {
764 for (auto pair : other)
765 {
766 remove(pair);
767 }
768 }

793 bool contains(const LeftType &left, const RightType &right) const noexcept
794 {
795 auto l2r_it = m_LeftToRight.find(left);
796 if (l2r_it != m_LeftToRight.end())
797 {
798 auto r2l_it = m_RightToLeft.find(right);

806 }
807 else
808 {
809 return 0 != countInSortedVector(r2l_vec, left);
810 }
811 }
812 }
813 return false;
814 }

990template <typename LeftType, typename RightType> class OneToOne
991{
992 std::unordered_map<LeftType, RightType> m_LeftToRight;
993 std::unordered_map<RightType, LeftType> m_RightToLeft;
994
995public:
999 struct Pair
1000 {
1001 LeftType left;
1002 RightType right;
1003 Pair(LeftType left, RightType right)
1004 : left(left), right(right)
1005 {}
1006 Pair()
1007 {}
1008 };
1009
1013 OneToOne() noexcept
1014 {}

1021 void insert(const Pair &pair) noexcept

1023 insert(pair.left, pair.right);
1024 }
1025
1032 void insert(const LeftType &left, const RightType &right) noexcept
1033 {
1034 if (!contains(left, right))
1035 {
1036 removeLeft(left);
1037 removeRight(right);
1038 m_RightToLeft[right] = left;
1039 m_LeftToRight[left] = right;
1040 }
1041 }

1064 }
1065
1072 void remove(const LeftType &left, const RightType &right) noexcept
1073 {
1074 if (contains(left, right))
1075 {
1076 auto l2r_it = m_LeftToRight.find(left);
1077 auto r2l_it = m_RightToLeft.find(right);
1078
1079 m_LeftToRight.erase(l2r_it);
1080 m_RightToLeft.erase(r2l_it);
1081 }
1082 }
1083
1089 void removeLeft(const LeftType &left) noexcept
1090 {
1091 auto l2r_it = m_LeftToRight.find(left);
1092 if (l2r_it != m_LeftToRight.end())
1093 {
