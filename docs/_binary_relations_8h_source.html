39template <typename T> bool containsInSortedVector(const std::vector<T> *vector, const T &value) noexcept
40{
41 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
42 return it != vector->cend() && *it == value;
43}
44
45template <typename T>
46typename std::vector<T>::const_iterator findInSortedVector(const std::vector<T> *vector, const T &value) noexcept
47{
48 typename std::vector<T>::const_iterator it = std::lower_bound(vector->cbegin(), vector->cend(), value);
49 if (it != vector->cend() && *it == value)
50 return it;
51 else
52 return vector->cend();
53}
54
55template <typename T> int insertIntoSortedVector(std::vector<T> *vector, const T &value) noexcept
56{
57 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
58 if (it != vector->end() && *it == value)
59 return 0; // It's already there
60 else
61 {
62 vector->insert(it, value);
63 return 1;
64 }
65}
66
67template <typename T> int eraseFromSortedVector(std::vector<T> *vector, const T &value) noexcept
68{
69 typename std::vector<T>::iterator it = std::lower_bound(vector->begin(), vector->end(), value);
70 if (it != vector->end() && *it == value)
71 {
72 vector->erase(it);
73 return 1;
74 }
75 else
76 return 0;
77}

79template <typename T>
80void insertIntoSortedVector(const std::vector<T> *sourceVector, const std::vector<T> *insertVector, std::vector<T> *outVector) noexcept
81{
82 typename std::vector<T>::const_iterator source_it = sourceVector->cbegin();
83 typename std::vector<T>::const_iterator insert_it = insertVector->cbegin();
84
85 outVector->clear();
86 outVector->reserve(sourceVector->size() + insertVector->size());
87
88 while (source_it != sourceVector->cend() && insert_it != insertVector->cend())
89 {
90 if (*source_it == *insert_it)
91 {
92 outVector->push_back(*source_it++);
93 insert_it++;
94 }
95 else if (*source_it < *insert_it)
96 {
97 outVector->push_back(*source_it++);
98 }
99 else
100 {
101 outVector->push_back(*insert_it++);
102 }
103 }
104
105 while (source_it != sourceVector->cend())
106 {
107 outVector->push_back(*source_it++);
108 }
109
110 while (insert_it != insertVector->cend())
111 {
112 outVector->push_back(*insert_it++);
113 }
114}

116template <typename T>
117void eraseFromSortedVector(const std::vector<T> *sourceVector, const std::vector<T> *eraseVector, std::vector<T> *outVector) noexcept
118{
119 typename std::vector<T>::const_iterator source_it = sourceVector->cbegin();
120 typename std::vector<T>::const_iterator erase_it = eraseVector->cbegin();
121 typename std::vector<T>::const_iterator source_end = sourceVector->cend();
122 typename std::vector<T>::const_iterator erase_end = eraseVector->cend();
123
124 outVector->clear();
125 outVector->reserve(std::max(sourceVector->size(), eraseVector->size()));
126
127 while (source_it != source_end && erase_it != erase_end)
128 {
129 if (*source_it == *erase_it)
130 {
131 source_it++;
132 erase_it++;
133 }
134 else if (*source_it < *erase_it)
135 {
136 outVector->push_back(*source_it++);
137 }
138 else
139 {
140 outVector->push_back(*erase_it++);
141 }
142 }
143
144 while (source_it != source_end)
145 {
146 outVector->push_back(*source_it++);
147 }
148}

211template <typename LeftType, typename RightType> class OneToMany
212{
213 std::unordered_map<LeftType, std::vector<RightType> *> m_LeftToRight;
214 std::unordered_map<RightType, LeftType> m_RightToLeft;
215 std::vector<RightType> m_EmptyRightVector;
216 bool m_RightIsSorted = false;
217
218public:
222 struct Pair
223 {
224 LeftType left;
225 RightType right;
226 Pair(LeftType left, RightType right)
227 : left(left), right(right)
228 {}
229 Pair()
230 {}
231 };
232
236 OneToMany() noexcept
237 {}

244 void insert(const Pair &pair) noexcept
245 {
246 insert(pair.left, pair.right);

255 void insert(const LeftType &left, const RightType &right) noexcept
256 {
257 auto r2l_it = m_RightToLeft.find(right);
258 if (r2l_it != m_RightToLeft.end())
259 {
260 if (r2l_it->second == left)
261 return; // We already have this pair - do nothing
262
263 erase(r2l_it->second, right); // Erase old relation
264 }
265
266 auto l2r_vec = m_LeftToRight[left]; // Will insert if it isn't already there.
267 if (l2r_vec == nullptr)
268 {
269 l2r_vec = new std::vector<RightType>();
270 m_LeftToRight[left] = l2r_vec;
271 }
272 insertIntoSortedVector(l2r_vec, right);
273
274 m_RightToLeft[right] = left;
275 }

283 void insert(const std::vector<Pair> &pairs) noexcept
284 {
285 if(0 == pairs.size())
286 return;
287
288 auto compare_left_then_right = [](const Pair &a, const Pair &b)
289 {
290 if(a.left < b.left) return true;
291 if(b.left < a.left) return false;
292 return a.right < b.right;
293 };
294
295 std::vector<Pair> pairs_to_insert = pairs; // Deep copy...
296 std::sort(pairs_to_insert.begin(), pairs_to_insert.end(), compare_left_then_right); // ...so I can sort
297
298 // ------------------------
299
300 std::vector<Pair> pairs_to_erase;
301 for (auto pair : pairs_to_insert)
302 {
303 auto r2l_it = m_RightToLeft.find(pair.right);
304 if (r2l_it != m_RightToLeft.end())
305 {
306 pairs_to_erase.push_back(Pair(r2l_it->second, r2l_it->first));
307 m_RightToLeft.erase(r2l_it);
308 }
309 }
310
311 if (0 != pairs_to_erase.size())
312 {
313 std::sort(pairs_to_erase.begin(), pairs_to_erase.end(), compare_left_then_right);
314
315 std::vector<RightType> right_to_erase;
316 auto end_it = pairs_to_erase.cend();
317 for (auto it = pairs_to_erase.cbegin(); it != end_it; )
318 {
319 // Collect all right values that have the same left value
320 right_to_erase.clear();
321 auto left = it->left;
322 while(it != end_it && it->left == left)
323 {
324 right_to_erase.push_back(it->right);
325 it++;
326 }
327
328 // Erase them in one go
329 auto l2r_it = m_LeftToRight.find(left);
330 if (l2r_it != m_LeftToRight.end())
331 {
332 auto l2r_vec = l2r_it->second;
333 eraseFromSortedVector(l2r_vec, &right_to_erase, l2r_vec);
334
335 if(0 == l2r_vec->size())
336 {
337 m_LeftToRight.erase(l2r_it);
338 delete l2r_vec;
339 }
340 }
341 }
342 }
343
344 // -----------------------
345
346 std::vector<RightType> right_to_insert;
347 auto it_end = pairs_to_insert.cend();
348 for (auto it = pairs_to_insert.cbegin(); it != it_end; )
349 {
350 // Collect all right values that have the same left value
351 right_to_insert.clear();
352 auto left = it->left;
353 while(it != it_end && it->left == left)
354 {
355 m_RightToLeft[it->right] = it->left;
356 right_to_insert.push_back(it->right);
357 it++;
358 }
359
360 // Insert them in one go
361 auto l2r_it = m_LeftToRight.find(left);
362 if (l2r_it != m_LeftToRight.end())
363 {
364 auto l2r_vec = l2r_it->second;
365 auto temp = *l2r_vec; // Deep copy
366 insertIntoSortedVector(&temp, &right_to_insert, l2r_vec);
367 }
368 else
369 {
370 // insert new value
371 auto l2r_vec = new std::vector<RightType>(right_to_insert);
372 m_LeftToRight[left] = l2r_vec;
373 }
374 }
375 }

393 void erase(const LeftType &left, const RightType &right) noexcept
394 {
395 auto r2l_it = m_RightToLeft.find(right);
396 if (r2l_it == m_RightToLeft.end())
397 return; // (*,right) not in the set - do nothing
398
399 if (r2l_it->second != left)
400 return; // (left,right) is not in the set - do nothing
401
402 auto l2r_it = m_LeftToRight.find(left);
403 auto l2r_vec = l2r_it->second;
404 auto l2r_vec_it = findInSortedVector(l2r_vec, right);
405 if (l2r_vec_it != l2r_vec->cend())
406 {
407 l2r_vec->erase(l2r_vec_it);
408 if (l2r_vec->size() == 0)
409 {
410 m_LeftToRight.erase(l2r_it);
411 delete l2r_vec; // Vector is empty now
412 }
413 m_RightToLeft.erase(r2l_it);
414 }
415 }

422 void eraseLeft(const LeftType &left) noexcept
423 {
424 auto l2r_it = m_LeftToRight.find(left);
425 if (l2r_it == m_LeftToRight.end())
426 return;
427
428 auto l2r_vec = l2r_it->second;
429 for (auto right : l2r_vec)
430 {
431 auto r2l_it = m_RightToLeft.find(right);
432 m_RightToLeft.erase(r2l_it);
433 }
434
435 m_LeftToRight.erase(l2r_it);
436 delete l2r_vec; // Vector is empty now
437 }

446 auto r2l_it = m_RightToLeft.find(right);
447 if (r2l_it == m_RightToLeft.end())
448 return;
449 erase(r2l_it->second, right);

633 class Iterator
634 {
636 public:
637 typename std::unordered_map<LeftType, std::vector<RightType> *>::const_iterator l2r_it;
638 typename std::unordered_map<LeftType, std::vector<RightType> *>::const_iterator l2r_it_end;
639 typename std::vector<RightType>::const_iterator l2r_vec_it;
640
641 inline Pair operator*() const noexcept
642 {
643 auto left = l2r_it->first;
644 auto right = *l2r_vec_it;
645 return Pair(left, right);
646 }
647
648 inline bool operator==(const Iterator &other) const noexcept
649 {
650 return l2r_it == other.l2r_it;
651 }
652
653 inline bool operator!=(const Iterator &other) const noexcept
654 {
655 return l2r_it != other.l2r_it;
656 }
657
658 inline Iterator operator++() noexcept
659 {
660 l2r_vec_it++;
661 if (l2r_vec_it == l2r_it->second->cend())
662 {
663 l2r_it++;
664 if (l2r_it != l2r_it_end)
665 {
666 l2r_vec_it = l2r_it->second->cbegin();
667 }
668 }
669 return *this;
670 }
672 };
673
678 Iterator begin() const noexcept
679 {
680 Iterator it;
681 it.l2r_it = m_LeftToRight.cbegin();
682 it.l2r_it_end = m_LeftToRight.cend();
683 if (it.l2r_it != it.l2r_it_end)
684 {
685 auto l2r_vec = it.l2r_it->second;
686 it.l2r_vec_it = l2r_vec->cbegin();
687 }
688 return it;
689 }
690
695 Iterator end() const noexcept
696 {
697 Iterator it;
698 it.l2r_it = m_LeftToRight.cend();
699 return it;
700 }

710{

717public:
721 struct Pair
722 {
723 LeftType left;
724 RightType right;

754 void insert(const LeftType &left, const RightType &right) noexcept
755 {
756 auto l2r_it = m_LeftToRight.find(left);
757 if (l2r_it != m_LeftToRight.end())
758 {
759 auto l2r_vec = l2r_it->second;
760 if (containsInSortedVector(l2r_vec, right))
761 return; // We already have this pair;
762 }
763
764 auto l2r_vec = m_LeftToRight[left]; // Will insert if it isn't already there.
765 if (l2r_vec == nullptr)
766 {
767 l2r_vec = new std::vector<RightType>();
768 m_LeftToRight[left] = l2r_vec;
769 }
770 insertIntoSortedVector(l2r_vec, right);
771
772 auto r2l_vec = m_RightToLeft[right]; // Will insert if it isn't already there.
773 if (r2l_vec == nullptr)
774 {
775 r2l_vec = new std::vector<LeftType>();
776 m_RightToLeft[right] = r2l_vec;
777 }
778 insertIntoSortedVector(r2l_vec, left);
779 m_Count += 1;
780 }

787 {

789 return;

793 if(a.left < b.left) return true;
794 if(b.left < a.left) return false;
795 return a.right < b.right;
796 };
797
798 auto compare_right_then_left = [](const Pair &a, const Pair &b)
799 {
800 if(a.right < b.right) return true;
801 if(b.right < a.right) return false;
802 return a.left < b.left;
803 };
804
805 std::vector<Pair> pairs_to_insert = pairs; // Deep copy...
806
807 // ------------------------
808
809 std::sort(pairs_to_insert.begin(), pairs_to_insert.end(), compare_left_then_right); // ...so I can sort
810
811 std::vector<RightType> right_to_insert;
812 auto it_end = pairs_to_insert.cend();
813 for (auto it = pairs_to_insert.cbegin(); it != it_end; )
814 {
815 // Collect all right values that have the same left value
816 right_to_insert.clear();
817 auto left = it->left;
818 while(it != it_end && it->left == left)
819 {
820 right_to_insert.push_back(it->right);
821 it++;
822 }
823
824 // Insert them in one go
825 auto l2r_it = m_LeftToRight.find(left);
826 if (l2r_it != m_LeftToRight.end())
827 {
828 auto l2r_vec = l2r_it->second;
829 auto temp = *l2r_vec;
830 m_Count -= l2r_vec->size();
831 insertIntoSortedVector(&temp, &right_to_insert, l2r_vec);
832 m_Count += l2r_vec->size();
833 }
834 else
835 {
836 // insert new value
837 auto l2r_vec = new std::vector<RightType>(right_to_insert);
838 m_LeftToRight[left] = l2r_vec;
839 m_Count += l2r_vec->size();
840 }
841 }
842
843 // ------------------------
844
845 std::sort(pairs_to_insert.begin(), pairs_to_insert.end(), compare_right_then_left);
846
847 std::vector<LeftType> left_to_insert;
848 it_end = pairs_to_insert.cend();
849 for (auto it = pairs_to_insert.cbegin(); it != it_end; )
850 {
851 // Collect all right values that have the same left value
852 left_to_insert.clear();
853 auto right = it->right;
854 while(it != it_end && it->right == right)
855 {
856 left_to_insert.push_back(it->left);
857 it++;
858 }
859
860 // Insert them in one go
861 auto r2l_it = m_RightToLeft.find(right);
862 if (r2l_it != m_RightToLeft.end())
863 {
864 auto r2l_vec = r2l_it->second;
865 auto temp = *r2l_vec;
866 insertIntoSortedVector(&temp, &left_to_insert, r2l_vec);
867 }
868 else
869 {
870 // insert new value
871 auto r2l_vec = new std::vector<LeftType>(left_to_insert);
872 m_RightToLeft[right] = r2l_vec;
873 }
874 }

893 void erase(const LeftType &left, const RightType &right) noexcept
894 {
895 auto l2r_it = m_LeftToRight.find(left);
896 if (l2r_it != m_LeftToRight.end())
897 {
898 auto r2l_it = m_RightToLeft.find(right);
899 if (r2l_it != m_RightToLeft.end())
900 {
901 auto l2r_vec = l2r_it->second;
902 auto r2l_vec = r2l_it->second;
903
904 eraseFromSortedVector(l2r_vec, right);
905 eraseFromSortedVector(r2l_vec, left);
906
907 if(l2r_vec->size() == 0)
908 {
909 m_LeftToRight.erase(l2r_it);
910 delete l2r_vec;
911 }
912
913 if(r2l_vec->size() == 0)
914 {
915 m_RightToLeft.erase(r2l_it);
916 delete r2l_vec;
917 }
918
919 m_Count -= 1;
920 }
921 }

929 void eraseLeft(const LeftType &left) noexcept
930 {
931 auto l2r_it = m_LeftToRight.constFind(left);
932 if (l2r_it != m_LeftToRight.end())
933 {
